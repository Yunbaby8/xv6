# 四、新增内核 syscall 实现文件

新建文件：

```bash
vim kernel/my_syscalls.c
```

完整内容如下：

```c
// kernel/my_syscalls.c

#include <linux/kernel.h>
#include <linux/syscalls.h>
#include <linux/sched.h>
#include <linux/uaccess.h>
#include <linux/errno.h>
#include <linux/string.h>

SYSCALL_DEFINE2(my_add, int, a, int, b)
{
	int result = a + b;

	printk(KERN_INFO "[my_add] current=%s pid=%d a=%d b=%d result=%d\n",
	       current->comm, current->pid, a, b, result);

	return result;
}

SYSCALL_DEFINE0(my_task_info)
{
	printk(KERN_INFO "[my_task_info] comm=%s pid=%d tgid=%d\n",
	       current->comm, current->pid, current->tgid);

	return current->pid;
}

SYSCALL_DEFINE2(my_get_comm, char __user *, buf, size_t, len)
{
	char comm[TASK_COMM_LEN];

	if (!buf) {
		printk(KERN_INFO "[my_get_comm] error: buf is NULL\n");
		return -EINVAL;
	}

	if (len < TASK_COMM_LEN) {
		printk(KERN_INFO "[my_get_comm] error: len=%zu is too small, need %d\n",
		       len, TASK_COMM_LEN);
		return -EINVAL;
	}

	get_task_comm(comm, current);

	if (copy_to_user(buf, comm, TASK_COMM_LEN)) {
		printk(KERN_INFO "[my_get_comm] error: copy_to_user failed\n");
		return -EFAULT;
	}

	printk(KERN_INFO "[my_get_comm] comm=%s pid=%d\n",
	       comm, current->pid);

	return 0;
}

SYSCALL_DEFINE2(my_print_user_msg, const char __user *, buf, size_t, len)
{
	char kbuf[64];

	if (!buf) {
		printk(KERN_INFO "[my_print_user_msg] error: buf is NULL\n");
		return -EINVAL;
	}

	if (len == 0 || len >= sizeof(kbuf)) {
		printk(KERN_INFO "[my_print_user_msg] error: invalid len=%zu\n", len);
		return -EINVAL;
	}

	memset(kbuf, 0, sizeof(kbuf));

	if (copy_from_user(kbuf, buf, len)) {
		printk(KERN_INFO "[my_print_user_msg] error: copy_from_user failed\n");
		return -EFAULT;
	}

	kbuf[len] = '\0';

	printk(KERN_INFO "[my_print_user_msg] comm=%s pid=%d msg=\"%s\"\n",
	       current->comm, current->pid, kbuf);

	return 0;
}
```

// test_syscall.c

#define _GNU_SOURCE
#include <unistd.h>
#include <sys/syscall.h>
#include <stdio.h>
#include <errno.h>
#include <string.h>

#define SYS_my_add             549
#define SYS_my_task_info       550
#define SYS_my_get_comm        551
#define SYS_my_print_user_msg  552

static void test_my_add(void)
{
	long ret;

	errno = 0;
	ret = syscall(SYS_my_add, 10, 20);

	if (ret < 0) {
		printf("[user] my_add failed: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] my_add ret = %ld\n", ret);
}

static void test_my_task_info(void)
{
	long ret;

	errno = 0;
	ret = syscall(SYS_my_task_info);

	if (ret < 0) {
		printf("[user] my_task_info failed: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] my_task_info ret pid = %ld\n", ret);
}

static void test_my_get_comm_ok(void)
{
	long ret;
	char comm[16];

	memset(comm, 0, sizeof(comm));

	errno = 0;
	ret = syscall(SYS_my_get_comm, comm, sizeof(comm));

	if (ret < 0) {
		printf("[user] my_get_comm failed: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] my_get_comm ret = %ld, comm = %s\n", ret, comm);
}

static void test_my_get_comm_small_len(void)
{
	long ret;
	char comm[4];

	memset(comm, 0, sizeof(comm));

	errno = 0;
	ret = syscall(SYS_my_get_comm, comm, sizeof(comm));

	if (ret < 0) {
		printf("[user] my_get_comm small len failed as expected: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] unexpected success: ret=%ld comm=%s\n", ret, comm);
}

static void test_my_get_comm_bad_pointer(void)
{
	long ret;

	errno = 0;
	ret = syscall(SYS_my_get_comm, (char *)0x1, 16);

	if (ret < 0) {
		printf("[user] my_get_comm bad pointer failed as expected: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] unexpected success: ret=%ld\n", ret);
}

static void test_my_print_user_msg_ok(void)
{
	long ret;
	const char *msg = "hello kernel";

	errno = 0;
	ret = syscall(SYS_my_print_user_msg, msg, strlen(msg));

	if (ret < 0) {
		printf("[user] my_print_user_msg failed: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] my_print_user_msg ret = %ld\n", ret);
}

static void test_my_print_user_msg_bad_pointer(void)
{
	long ret;

	errno = 0;
	ret = syscall(SYS_my_print_user_msg, (char *)0x1, 12);

	if (ret < 0) {
		printf("[user] my_print_user_msg bad pointer failed as expected: ret=%ld errno=%d %s\n",
		       ret, errno, strerror(errno));
		return;
	}

	printf("[user] unexpected success: ret=%ld\n", ret);
}

int main(void)
{
	printf("===== syscall lab test start =====\n");

	test_my_add();
	test_my_task_info();
	test_my_get_comm_ok();
	test_my_get_comm_small_len();
	test_my_get_comm_bad_pointer();
	test_my_print_user_msg_ok();
	test_my_print_user_msg_bad_pointer();

	printf("===== syscall lab test end =====\n");

	return 0;
}

# 十三、如果编译报 syscall 声明冲突

如果你遇到类似：

```text
conflicting types for sys_my_add
```

一般是 `include/linux/syscalls.h` 里的声明和 `SYSCALL_DEFINE` 自动生成的类型不匹配。

你可以先把 `include/linux/syscalls.h` 中新增的 4 行删掉试试。

在很多新内核里，使用 `SYSCALL_DEFINE` 并配合 syscall 表，通常不一定需要你手动声明这几个原型。

也就是说，如果报声明冲突，保留：

```c
SYSCALL_DEFINE2(my_add, int, a, int, b)
SYSCALL_DEFINE0(my_task_info)
SYSCALL_DEFINE2(my_get_comm, char __user *, buf, size_t, len)
SYSCALL_DEFINE2(my_print_user_msg, const char __user *, buf, size_t, len)
```

然后删除 `include/linux/syscalls.h` 里你加的：

```c
asmlinkage long sys_my_add(int a, int b);
asmlinkage long sys_my_task_info(void);
asmlinkage long sys_my_get_comm(char __user *buf, size_t len);
asmlinkage long sys_my_print_user_msg(const char __user *buf, size_t len);
```

重新编译。
